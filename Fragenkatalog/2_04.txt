#include<stdio.h>
#include<stdlib.h>
#include<limits.h> //fÃ¼r MAKRO ULONG_MAX

unsigned long safe_add(unsigned long a, unsigned long b){
  if(a == 0){ // eigentlich optional
    return b;
  }
  if(b == 0){ // eigentlich optional
    return a;
  }
  if(a > ULONG_MAX - b){
    frptinf(stderr, "%s overflow(%lu + %lu)\n", __func__, a, b);
    exit(EXIT_FAILURE);
  }
  return a+b;
}
