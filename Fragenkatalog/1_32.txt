long sum_of_lines(const char *progname, const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp==NULL) {
        fprintf(stderr, "%s: cannot open file %s\n", progname, filename);
        return EXIT_FAILURE;
    }
    char buff[50]; //Ähm, wahrscheinlich nicht die effizienteste Lösung, aber sie funktioniert (getestet)
	long summe = 0;
	long a, b, c;
	while(fgets(buff, 50, fp)!=NULL) {
		sscanf(buff, "%ld %ld %ld", &a, &b, &c);
		summe += (a+b+c);
	}
    fclose(fp);
	return summe;
}
